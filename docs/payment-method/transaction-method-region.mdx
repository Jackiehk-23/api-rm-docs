import Admonition from '@theme/Admonition';
import payments from "../../src/config/payment-method.js"
import { FaCheck } from "react-icons/fa"
import MarkdownTable from '@site/src/components/MarkdownTable'
import PayIcon from '@site/src/components/PayIcon'

export const Check = ({ label }) => (
  <span className="rm-tip" data-tip={label}>
    <FaCheck color="#22c55e" />
  </span>
);

<Admonition type="note">
**For online web and mobile web payments.** Append the region code to the system method name. For example, Malaysia uses `MY`, so the full method name becomes `ALIPAYPLUS_MY`.
</Admonition>

### Method Availability Table

<MarkdownTable>
  <thead>
    <tr>
      <th>Method</th>
      <th>Region</th>
      <th>Quick Pay</th>
      <th>Web Payment</th>
      <th>Mobile Web Payment</th>
      <th>Transaction QR</th>
      <th>Retail QR</th>
    </tr>
  </thead>
  <tbody>
    {payments
      .slice()
      .sort((a, b) => a.method.localeCompare(b.method) || a.region.localeCompare(b.region))
      .map((payment, index) => (
      <tr key={index}>
        <td>
          <div style={{display:'flex',alignItems:'center',gap:'8px'}}>
            <PayIcon path={payment.logo} size={28} alt={payment.method} />
            <span>{payment.method}</span>
          </div>
        </td>
        <td>{payment.region}</td>
        <td style={{textAlign: 'center'}}>{payment.quickPay && <Check label="Quick Pay" />}</td>
        <td style={{textAlign: 'center'}}>{payment.web && <Check label="Web Payment" />}</td>
        <td style={{textAlign: 'center'}}>{payment.mobileWeb && <Check label="Mobile Web Payment" />}</td>
        <td style={{textAlign: 'center'}}>{payment.transactionQR && <Check label="Transaction QR" />}</td>
        <td style={{textAlign: 'center'}}>{payment.retailQR && <Check label="Retail QR" />}</td>
      </tr>
    ))}
  </tbody>
</MarkdownTable>
