import Admonition from '@theme/Admonition';
import bankCodes from '../src/config/bank-codes.js';
import MarkdownTable from '@site/src/components/MarkdownTable'
import PayIcon from '@site/src/components/PayIcon'

export const CodeBadge = ({ code, label }) => (
  <div style={{display:'inline-flex',alignItems:'center',gap:'8px',marginBottom:'4px'}}>
    <span style={{
      display:'inline-block',
      padding:'3px 8px',
      borderRadius:'5px',
      fontSize:'12px',
      fontFamily:'monospace',
      fontWeight:600,
      background:'#f1f5f9',
      color:'#334155',
      border:'1px solid #cbd5e1',
      whiteSpace:'nowrap',
    }}>
      {code}
    </span>
    {label && (
      <span style={{fontSize:'12px',color:'var(--ifm-color-emphasis-600)',whiteSpace:'nowrap'}}>{label}</span>
    )}
  </div>
);

export const BankCell = ({ logo, name }) => (
  <div style={{display:'flex',alignItems:'center',gap:'10px'}}>
    <PayIcon path={logo} size={40} alt={name} />
    <span style={{fontWeight:500}}>{name}</span>
  </div>
);

Bank codes supported by Revenue Monster for FPX payments.

<Admonition type="note">
Use these codes when integrating with FPX-based payment methods. B2C codes are for personal banking; B2B codes are for business/corporate banking.
</Admonition>

<MarkdownTable>
  <thead>
    <tr>
      <th rowSpan={2} style={{verticalAlign:'middle'}}>Bank</th>
      <th colSpan={2} style={{textAlign:'center'}}>Bank Code</th>
    </tr>
    <tr>
      <th>B2C (Personal)</th>
      <th>B2B (Business)</th>
    </tr>
  </thead>
  <tbody>
    {bankCodes.flatMap(bank => {
      // Group this bank's entries by portal name so each distinct portal
      // (e.g. "Affin Bank" vs "AFFINMAX") gets its own row.
      const order = [];
      const byName = {};
      bank.entries.forEach(e => {
        if (!byName[e.name]) { byName[e.name] = { b2c: null, b2b: null }; order.push(e.name); }
        if (e.code.includes(':B2C')) byName[e.name].b2c = e.code;
        else if (e.code.includes(':B2B')) byName[e.name].b2b = e.code;
      });
      return order.map(name => (
        <tr key={bank.name + '|' + name}>
          <td style={{verticalAlign:'middle'}}>
            <BankCell logo={bank.logo} name={name} />
          </td>
          <td style={{verticalAlign:'middle'}}>
            {byName[name].b2c
              ? <CodeBadge code={byName[name].b2c} />
              : <span style={{color:'#94a3b8',fontSize:'13px'}}>—</span>}
          </td>
          <td style={{verticalAlign:'middle'}}>
            {byName[name].b2b
              ? <CodeBadge code={byName[name].b2b} />
              : <span style={{color:'#94a3b8',fontSize:'13px'}}>—</span>}
          </td>
        </tr>
      ));
    })}
  </tbody>
</MarkdownTable>
