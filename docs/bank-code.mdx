import Admonition from '@theme/Admonition';
import bankCodes from '../src/config/bank-codes.js';
import MarkdownTable from '@site/src/components/MarkdownTable'

export const CodeBadge = ({ code, isOnline }) => (
  <div style={{display:'inline-flex',alignItems:'center',gap:'6px',marginBottom:'4px'}}>
    <span style={{
      display:'inline-block',
      padding:'3px 8px',
      borderRadius:'5px',
      fontSize:'12px',
      fontFamily:'monospace',
      fontWeight:600,
      background:'#f1f5f9',
      color:'#334155',
      border:'1px solid #cbd5e1',
      whiteSpace:'nowrap',
    }}>
      {code}
    </span>
    {!isOnline && (
      <span style={{
        display:'inline-block',
        padding:'2px 6px',
        borderRadius:'4px',
        fontSize:'11px',
        fontWeight:500,
        background:'#fef3c7',
        color:'#92400e',
        border:'1px solid #fde68a',
      }}>Offline</span>
    )}
  </div>
);

export const BankCell = ({ logo, name }) => (
  <div style={{display:'flex',alignItems:'center',gap:'10px'}}>
    {logo
      ? <img src={logo} style={{height:'40px',width:'40px',objectFit:'contain',flexShrink:0}} />
      : <span style={{display:'inline-block',width:'40px',height:'40px',flexShrink:0,background:'#e2e8f0',borderRadius:'50%'}} />
    }
    <span style={{fontWeight:500}}>{name}</span>
  </div>
);

Bank codes supported by Revenue Monster for FPX payments.

<Admonition type="note">
Use these codes when integrating with FPX-based payment methods. B2C codes are for personal banking; B2B codes are for business/corporate banking.
</Admonition>

<MarkdownTable>
  <thead>
    <tr>
      <th rowSpan={2} style={{verticalAlign:'middle'}}>Bank</th>
      <th colSpan={2} style={{textAlign:'center'}}>Bank Code</th>
    </tr>
    <tr>
      <th>B2C (Personal)</th>
      <th>B2B (Business)</th>
    </tr>
  </thead>
  <tbody>
    {bankCodes.map(bank => {
      const b2c = bank.entries.filter(e => e.code.includes(':B2C'));
      const b2b = bank.entries.filter(e => e.code.includes(':B2B'));
      return (
        <tr key={bank.name}>
          <td style={{verticalAlign:'middle'}}>
            <BankCell logo={bank.logo} name={bank.name} />
          </td>
          <td style={{verticalAlign:'middle'}}>
            <div style={{display:'flex',flexDirection:'column',gap:'4px'}}>
              {b2c.length > 0
                ? b2c.map(e => <CodeBadge key={e.code} code={e.code} isOnline={e.isOnline} />)
                : <span style={{color:'#94a3b8',fontSize:'13px'}}>—</span>
              }
            </div>
          </td>
          <td style={{verticalAlign:'middle'}}>
            <div style={{display:'flex',flexDirection:'column',gap:'4px'}}>
              {b2b.length > 0
                ? b2b.map(e => <CodeBadge key={e.code} code={e.code} isOnline={e.isOnline} />)
                : <span style={{color:'#94a3b8',fontSize:'13px'}}>—</span>
              }
            </div>
          </td>
        </tr>
      );
    })}
  </tbody>
</MarkdownTable>
