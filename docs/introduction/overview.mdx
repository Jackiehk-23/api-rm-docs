import React, { useState } from 'react';
import { LuZap, LuBuilding2, LuCreditCard, LuGift, LuPackage, LuScanFace, LuMail, LuHeadphones, LuLock, LuInfo } from "react-icons/lu";
import { TbCircleNumber1, TbCircleNumber2, TbCircleNumber3 } from "react-icons/tb";
import Admonition from '@theme/Admonition';

export const CopyUrl = ({children}) => {
  const [ok, setOk] = useState(false);
  return (
    <div
      onClick={() => { navigator.clipboard.writeText(children); setOk(true); setTimeout(() => setOk(false), 1800); }}
      style={{
        display:'flex', alignItems:'center', justifyContent:'space-between',
        background:'var(--ifm-color-emphasis-100)', border:'1px solid var(--ifm-color-emphasis-200)',
        borderRadius:'6px', padding:'8px 12px', marginBottom:'10px', cursor:'pointer',
      }}
    >
      <code style={{fontFamily:'var(--ifm-font-family-monospace)',fontSize:'0.78rem',background:'none',padding:0,border:'none',color:'var(--ifm-font-color-base)'}}>{children}</code>
      <span style={{
        marginLeft:'10px', padding:'2px 8px', borderRadius:'4px', flexShrink:0,
        border:`1px solid ${ok ? 'rgba(104,211,145,0.4)' : 'var(--ifm-color-emphasis-300)'}`,
        background: ok ? 'rgba(104,211,145,0.12)' : 'var(--ifm-color-emphasis-200)',
        color: ok ? '#68d391' : 'var(--ifm-color-emphasis-600)',
        fontSize:'10px', fontFamily:'monospace', transition:'all 0.2s',
      }}>{ok ? '✓' : 'Copy'}</span>
    </div>
  );
};
