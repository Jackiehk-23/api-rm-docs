export const CopyUrl = ({children}) => {
  const [ok, setOk] = React.useState(false);
  return (
    <div
      onClick={() => { navigator.clipboard.writeText(children); setOk(true); setTimeout(() => setOk(false), 1800); }}
      style={{
        display:'flex', alignItems:'center', justifyContent:'space-between',
        background:'var(--ifm-code-background)', borderRadius:'6px',
        padding:'8px 12px', marginBottom:'10px', cursor:'pointer',
      }}
    >
      <code style={{fontFamily:'var(--ifm-font-family-monospace)',fontSize:'0.78rem',background:'none',padding:0,border:'none'}}>{children}</code>
      <span style={{
        marginLeft:'10px', padding:'2px 8px', borderRadius:'4px', flexShrink:0,
        border:`1px solid ${ok ? 'rgba(104,211,145,0.4)' : 'var(--ifm-color-emphasis-300)'}`,
        background: ok ? 'rgba(104,211,145,0.12)' : 'var(--ifm-color-emphasis-200)',
        color: ok ? '#68d391' : 'var(--ifm-color-emphasis-600)',
        fontSize:'10px', fontFamily:'monospace', transition:'all 0.2s',
      }}>{ok ? '✓' : 'Copy'}</span>
    </div>
  );
};
