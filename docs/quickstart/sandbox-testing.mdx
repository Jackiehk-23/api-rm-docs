import React from 'react';
import Admonition from '@theme/Admonition';
import { LuCreditCard, LuLandmark, LuWallet, LuQrCode } from "react-icons/lu";

export const CopyValue = ({ children, copy }) => {
  const [copied, setCopied] = React.useState(false);
  const text = copy ?? (typeof children === 'string' ? children : String(children));
  return (
    <button
      type="button"
      className={`sbx-code sbx-copy ${copied ? 'sbx-copied' : ''}`}
      onClick={() => {
        navigator.clipboard.writeText(text).then(() => {
          setCopied(true);
          setTimeout(() => setCopied(false), 1500);
        });
      }}
      title="Click to copy"
      aria-label="Copy card number"
    >
      <span>{children}</span>
      <span className="sbx-copy-icon" aria-hidden="true">{copied ? '✓' : '⧉'}</span>
    </button>
  );
};
